  1. Outlier Analysis (IQR)

Q1 = df['sepal length'].quantile(0.25)
Q3 = df['sepal length'].quantile(0.75)
IQR = Q3 - Q1
outliers = df[(df['sepal length'] < Q1 - 1.5*IQR) |
                       (df['sepal length'] > Q3 + 1.5*IQR)]
print(outliers)
